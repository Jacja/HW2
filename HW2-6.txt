import random

n = 10
print('Загадано целое число от 0 до 100.')
print(f'Необходимо угадать его за {n} попыток.')

number = random.randrange(0, 101)


while n > 0:
    you_num = int(input('Введите Ваше число : '))
    if you_num == number:
        print('Вы победили!')
        break
    else:
        if you_num > number:
            print('Ваше число больше загаданого.')
        else:
            print('Ваше число меньше загаданного.')
        n -= 1
        print(f'попыток осталось : {n}')
        if n == 0:
            print(f'Вы не смогли угадать число: {number}, попробуйте в следующий раз.')


В программе генерируется случайное целое число от 0 до 100. Пользователь должен его отгадать не более чем за 10 попыток. После каждой неудачной попытки должно сообщаться больше или меньше введенное пользователем число, чем то, что загадано. Если за 10 попыток число не отгадано, то вывести загаданное число.
